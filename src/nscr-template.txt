&&header
*define

nsa
globalon

; text
numalias SP_TEXT_SP_BEGIN,4
numalias SP_TEXT_SP_END,8

numalias VARLT_TEXTBOX_NEXT,27

defsub text
defsub text_off

; title_dsp

numalias SP_TITLE_TEXT,2
numalias VARLT_TITLE_CUR,28

effect 2,10,500

defsub title_dsp

; textbox
numalias SP_TEXTBOX_NAMEBOX_IMG,1
numalias SP_TEXTBOX_NAMEBOX_TEXT,0

numalias VARLT_TEXTBOX_NAMEBOX_CURNAME,24
numalias VARLT_TEXTBOX_NAMEBOX_X,25
numalias VARLT_TEXTBOX_NAMEBOX_Y,26

defsub textbox_set_name
defsub textbox_clear_name

; compiled define

&&defines

game

*start

mov %VARLT_TEXTBOX_NEXT,SP_TEXT_SP_BEGIN

&&boot
&&body

*textbox_set_name
getparam $VARLT_TEXTBOX_NAMEBOX_CURNAME
mov %VARLT_TEXTBOX_NAMEBOX_CURNAME,1
gosub *textbox_show
return

*textbox_clear_name
mov $VARLT_TEXTBOX_NAMEBOX_CURNAME,""
mov %VARLT_TEXTBOX_NAMEBOX_CURNAME,0
gosub *textbox_show
return

*textbox_show
lsp SP_TEXTBOX_NAMEBOX_TEXT,":s;#ffffff"+$VARLT_TEXTBOX_NAMEBOX_CURNAME,0,0
getspsize SP_TEXTBOX_NAMEBOX_IMG,%0,%1
getspsize SP_TEXTBOX_NAMEBOX_TEXT,%2,%3
mov %4,%1
div %4,2
add %4,%VARLT_TEXTBOX_NAMEBOX_Y
mov %5,%3
div %5,2
sub %4,%5
if CONST_PYMO_GAMECONFIG_NAMEALIGN==0 mov %5,%VARLT_TEXTBOX_NAMEBOX_X:add %5,4
if CONST_PYMO_GAMECONFIG_NAMEALIGN==1 mov %5,%VARLT_TEXTBOX_NAMEBOX_X:add %5,%0:sub %5,%2:sub %5,4
if CONST_PYMO_GAMECONFIG_NAMEALIGN==2 gosub *textbox_calc_name_text_pos_middle
amsp SP_TEXTBOX_NAMEBOX_TEXT,%5,%4
vsp SP_TEXTBOX_NAMEBOX_IMG,%VARLT_TEXTBOX_NAMEBOX_CURNAME
vsp SP_TEXTBOX_NAMEBOX_TEXT,%VARLT_TEXTBOX_NAMEBOX_CURNAME
print 1
return

*textbox_calc_name_text_pos_middle
mov %5,%0
sub %5,%2
div %5,2
add %5,%VARLT_TEXTBOX_NAMEBOX_X
return

*textbox_hide
vsp SP_TEXTBOX_NAMEBOX_IMG,0
vsp SP_TEXTBOX_NAMEBOX_TEXT,0
print 1
return

*text
getparam $0,%1,%2
lsp %VARLT_TEXTBOX_NEXT,$0,%1,%2
inc %VARLT_TEXTBOX_NEXT
if %VARLT_TEXTBOX_NEXT==SP_TEXT_SP_END mov %VARLT_TEXTBOX_NEXT,SP_TEXT_SP_BEGIN
print 1
return

*text_off
mov %VARLT_TEXTBOX_NEXT,SP_TEXT_SP_BEGIN
for %0=SP_TEXT_SP_BEGIN to SP_TEXT_SP_END
csp %0
next
print 1
return

*title_dsp
lsp SP_TITLE_TEXT,":s/32,32,0;#ffffff"+$VARLT_TITLE_CUR,100,50
print 2
btntime 5000
btnwait %0
csp SP_TITLE_TEXT
print 2
return
